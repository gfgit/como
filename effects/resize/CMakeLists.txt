)

set(kwin_resize_config_SRCS resize_config.cpp)

    KF5::ConfigWidgets

    Qt::DBus
    KWinEffectsInterface
)

install(
    TARGETS
        kwin_resize_config
    DESTINATION
        ${PLUGIN_INSTALL_DIR}/kwin/effects/configs
